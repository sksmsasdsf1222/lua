-- 곡선 모서리 적용 함수
local function setRoundedCorners(instance, radius)
    instance.ClipsDescendants = true
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, radius)
    corner.Parent = instance
end

-- X 버튼 생성 함수
local function createCloseButton(parent)
    local closeButton = Instance.new("TextButton")
    closeButton.Size = UDim2.new(0, 30, 0, 30)
    closeButton.Position = UDim2.new(1, -40, 0, 10)
    closeButton.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
    closeButton.BorderSizePixel = 0
    closeButton.Text = "✕"
    closeButton.TextColor3 = Color3.fromRGB(255, 255, 255)
    closeButton.Font = Enum.Font.GothamBold
    closeButton.TextSize = 18
    closeButton.TextStrokeTransparency = 0.8
    closeButton.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
    setRoundedCorners(closeButton, 10)
    closeButton.Parent = parent

    closeButton.MouseButton1Click:Connect(function()
        parent:TweenSizeAndPosition(
            UDim2.new(0, 0, 0, 0),
            UDim2.new(0.5, 0, 0.5, 0),
            Enum.EasingDirection.In,
            Enum.EasingStyle.Sine,
            0.5,
            true,
            function() parent:Destroy() end
        )
    end)
    
    return closeButton
end

    errorNotification.Size = UDim2.new(0, 0, 0, 0)
    errorNotification.Position = UDim2.new(0.5, 0, 0, -50)
    errorNotification.AnchorPoint = Vector2.new(0.5, 0)
    errorNotification.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
    errorNotification.BorderSizePixel = 0
    setRoundedCorners(errorNotification, 20)
    errorNotification.Parent = parentFrame

    local contentLabel = Instance.new("TextLabel")
    contentLabel.Size = UDim2.new(1, 0, 1, 0)
    contentLabel.BackgroundTransparency = 1
    contentLabel.Text = "비밀번호를 틀렸습니다."
    contentLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
    contentLabel.Font = Enum.Font.Gotham
    contentLabel.TextSize = 18
    contentLabel.TextStrokeTransparency = 0.8
    contentLabel.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
    contentLabel.Parent = errorNotification

    setRoundedCorners(espButton, 10)

    -- 날기 기능 활성화
    local function toggleFly()
        local char = player.Character
        if not char then return end
        local humanoid = char:FindFirstChildOfClass("Humanoid")
        if not humanoid then return end

        if not char:FindFirstChild("Fly") then
            local flyScript = Instance.new("LocalScript")
            flyScript.Name = "Fly"
            flyScript.Source = [[
                local player = game.Players.LocalPlayer
                local mouse = player:GetMouse()
                local char = player.Character
                local flying = false

                local function fly()
                    flying = true
                    while flying do
                        local direction = (mouse.Hit.p - char.PrimaryPart.Position).unit
                        char.PrimaryPart.Velocity = direction * 50
                        wait(0.1)
                    end
                end

                mouse.KeyDown:Connect(function(key)
                    if key == "space" then
                        if flying then
                            flying = false
                            char.PrimaryPart.Velocity = Vector3.new(0, 0, 0)
                        else
                            fly()
                        end
                    end
                end)
            ]]
            flyScript.Parent = char
        else
            char:FindFirstChild("Fly"):Destroy()
        end
    end

    -- ESP 기능 활성화
    local function toggleESP()
        local function createESP(player)
            local espPart = Instance.new("Part")
            espPart.Size = Vector3.new(4, 1, 4)
            espPart.Anchored = true
            espPart.CanCollide = false
            espPart.BrickColor = BrickColor.new("Bright red")
            espPart.Material = Enum.Material.SmoothPlastic
            espPart.Transparency = 0.5
            espPart.Parent = player.Character
            espPart.CFrame = player.Character.HumanoidRootPart.CFrame + Vector3.new(0, 5, 0)

            local nameTag = Instance.new("BillboardGui")
            nameTag.Size = UDim2.new(0, 200, 0, 50)
            nameTag.Adornee = espPart
            nameTag.Parent = espPart

            local label = Instance.new("TextLabel")
            label.Size = UDim2.new(1, 0, 1, 0)
            label.BackgroundTransparency = 1
            label.Text = player.Name
            label.TextColor3 = Color3.fromRGB(255, 255, 255)
            label.TextStrokeTransparency = 0.8
            label.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
            label.Font = Enum.Font.GothamBold
            label.TextSize = 20
            label.Parent = nameTag
        end

        for _, p in ipairs(Players:GetPlayers()) do
            if p ~= player then
                if not p.Character:FindFirstChild("ESP") then
                    createESP(p)
                end
            end
        end
    end

    -- 버튼 클릭 시 이벤트 처리
    flyButton.MouseButton1Click:Connect(toggleFly)
    espButton.MouseButton1Click:Connect(toggleESP)
end

-- 프로그램 실행
local function start()
    local passwordScreen, passwordBox, submitButton, mainFrame = createPasswordScreen()
    
    submitButton.MouseButton1Click:Connect(function()
        if passwordBox.Text == "correctPassword" then
            passwordScreen:Destroy()
            createImportantSystemGui()
        else
            createErrorNotification(mainFrame)
        end
    end)
end

start()
