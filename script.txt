local Players = game:GetService("Players")
local player = Players.LocalPlayer
local playerGui = player:WaitForChild("PlayerGui")
local TweenService = game:GetService("TweenService")

-- 곡선 모서리 적용 함수
local function setRoundedCorners(frame, radius)
    frame.ClipsDescendants = true
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, radius)
    corner.Parent = frame
end

-- X 버튼 생성 함수
local function createCloseButton(parent)
    local closeButton = Instance.new("TextButton")
    closeButton.Size = UDim2.new(0, 30, 0, 30)
    closeButton.Position = UDim2.new(1, -40, 0, 10)
    closeButton.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
    closeButton.BorderSizePixel = 0
    closeButton.Text = "✕"
    closeButton.TextColor3 = Color3.fromRGB(255, 255, 255)
    closeButton.Font = Enum.Font.GothamBold
    closeButton.TextSize = 18
    closeButton.TextStrokeTransparency = 0.8
    closeButton.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
    setRoundedCorners(closeButton, 10)
    closeButton.Parent = parent

    closeButton.MouseButton1Click:Connect(function()
        parent:TweenSizeAndPosition(
            UDim2.new(0, 0, 0, 0),
            UDim2.new(0.5, 0, 0.5, 0),
            Enum.EasingDirection.In,
            Enum.EasingStyle.Sine,
            0.5,
            true,
            function() parent:Destroy() end
        )
    end)
    
    return closeButton
end

-- 중요 시스템 GUI 생성 함수 (업그레이드 버전)
local function createImportantSystemGui()
    local systemGui = Instance.new("ScreenGui")
    systemGui.Name = "ImportantSystemGui"
    systemGui.Parent = playerGui

    local systemFrame = Instance.new("Frame")
    systemFrame.Size = UDim2.new(0, 0, 0, 0)
    systemFrame.Position = UDim2.new(0.5, 0, 0.5, 0)
    systemFrame.AnchorPoint = Vector2.new(0.5, 0.5)
    systemFrame.BackgroundColor3 = Color3.fromRGB(35, 35, 35)  -- 더 어두운 배경색
    systemFrame.BorderSizePixel = 0
    setRoundedCorners(systemFrame, 20)  -- 곡선 모서리 적용
    systemFrame.Parent = systemGui

    -- 창 확장 애니메이션
    local tween = TweenService:Create(systemFrame, TweenInfo.new(0.5, Enum.EasingStyle.Bounce, Enum.EasingDirection.Out), {Size = UDim2.new(0.6, 0, 0.6, 0)})

    titleFrame.Size = UDim2.new(1, 0, 0, 50)
    titleFrame.BackgroundColor3 = Color3.fromRGB(65, 65, 65)
    titleFrame.BorderSizePixel = 0
    setRoundedCorners(titleFrame, 20)  -- 곡선 모서리 적용
    titleFrame.Parent = systemFrame

    -- 제목 텍스트
    local titleLabel = Instance.new("TextLabel")
    titleLabel.Size = UDim2.new(1, 0, 1, 0)
    titleLabel.BackgroundTransparency = 1
    titleLabel.Text = "[ KR ] PRO ViSNT HUB"
    titleLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
    titleLabel.Font = Enum.Font.GothamBold
    titleLabel.TextSize = 24
    titleLabel.TextStrokeTransparency = 0.8
    titleLabel.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
    titleLabel.Parent = titleFrame

    -- 컨텐츠 레이블 (텍스트)
    local contentLabel = Instance.new("TextLabel")
    contentLabel.Size = UDim2.new(0.9, 0, 0.6, 0)
    contentLabel.Position = UDim2.new(0.5, 0, 0.55, 0)
    contentLabel.AnchorPoint = Vector2.new(0.5, 0.5)
    contentLabel.BackgroundTransparency = 1
    contentLabel.Text = "환영합니다! 이 창은 시스템 정보를 표시합니다."
    contentLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
    contentLabel.Font = Enum.Font.Gotham
    contentLabel.TextSize = 20
    contentLabel.TextWrapped = true
    contentLabel.TextStrokeTransparency = 0.8
    contentLabel.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
    contentLabel.Parent = systemFrame

    -- 확인 버튼
    local confirmButton = Instance.new("TextButton")
    confirmButton.Size = UDim2.new(0.3, 0, 0, 45)
    confirmButton.Position = UDim2.new(0.5, 0, 0.85, 0)
    confirmButton.AnchorPoint = Vector2.new(0.5, 0.5)
    confirmButton.BackgroundColor3 = Color3.fromRGB(0, 122, 255)  -- 밝은 파란색
    confirmButton.BorderSizePixel = 0
    confirmButton.Text = "확인"
    confirmButton.TextColor3 = Color3.fromRGB(255, 255, 255)
    confirmButton.Font = Enum.Font.GothamBold
    confirmButton.TextSize = 20
    confirmButton.TextStrokeTransparency = 0.8
    confirmButton.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
    setRoundedCorners(confirmButton, 10)  -- 곡선 모서리 적용
    confirmButton.Parent = systemFrame

    confirmButton.MouseButton1Click:Connect(function()
        systemFrame:TweenSizeAndPosition(
            UDim2.new(0, 0, 0, 0),
            UDim2.new(0.5, 0, 0.5, 0),
            Enum.EasingDirection.In,
            Enum.EasingStyle.Sine,
            0.5,
            true,
            function() systemGui:Destroy() end
        )
    end)
end

-- 비밀번호 입력과 알림 창 설정
local function setup()
    local passwordScreen, passwordBox, submitButton, mainFrame = createPasswordScreen()

    -- 비밀번호가 맞는지 확인할 예제 함수 (사용자의 비밀번호 입력 확인)
    local correctPassword = "1234"  -- 예제 비밀번호
    submitButton.MouseButton1Click:Connect(function()
        if passwordBox.Text == correctPassword then
            passwordScreen:Destroy()
            createImportantSystemGui()
        else
            createErrorNotification(mainFrame)
        end
    end)
end

setup()
