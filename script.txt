-- 곡선 모서리 적용 함수
local function setRoundedCorners(instance, radius)
    instance.ClipsDescendants = true
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, radius)
    corner.Parent = instance
end

local function createCloseButton(parent)
    local closeButton = Instance.new("TextButton")
    closeButton.Size = UDim2.new(0, 30, 0, 30)
    closeButton.Position = UDim2.new(1, -40, 0, 10)
    closeButton.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
    closeButton.BorderSizePixel = 0
    closeButton.Text = "✕"
    closeButton.TextColor3 = Color3.fromRGB(255, 255, 255)
    closeButton.Font = Enum.Font.GothamBold
    closeButton.TextSize = 18
    closeButton.TextStrokeTransparency = 0.8
    closeButton.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
    setRoundedCorners(closeButton, 10)
    closeButton.Parent = parent

    closeButton.MouseButton1Click:Connect(function()
        parent:TweenSizeAndPosition(
            UDim2.new(0, 0, 0, 0),
            UDim2.new(0.5, 0, 0.5, 0),
            Enum.EasingDirection.In,
            Enum.EasingStyle.Sine,
            0.5,
            true,
            function() parent:Destroy() end
        )
    end)
    
    return closeButton
end

-- ESP 기능
local function createESP()
    local Settings = {
        ['Material'] = Enum.Material.Neon, -- Material
        ['Color'] = Color3.fromRGB(0, 255, 255), -- Color
        ['Transparency'] = 0.7 -- 0 to 1 Transparency
    }

    local ScreenGui = Instance.new('ScreenGui', game.CoreGui) -- Create screengui
    ScreenGui.IgnoreGuiInset = true

    local ViewportFrame = Instance.new('ViewportFrame', ScreenGui) -- Create viewport and define properties
    ViewportFrame.CurrentCamera = workspace.CurrentCamera
    ViewportFrame.Size = UDim2.new(1, 0, 1, 0)
    ViewportFrame.BackgroundTransparency = 1
    ViewportFrame.ImageTransparency = Settings.Transparency

    local Chasms = {} -- Array for storing parts

    local function generateChasm(player) -- functions that generates chasms for the player specififed
        local Character = workspace:FindFirstChild(player.Name)
        if Character then
            for _, Part in pairs(Character:GetChildren()) do
                if Part:IsA('Part') or Part:IsA('MeshPart') then
                    local Chasm = Part:Clone()
                    for _, Child in pairs(Chasm:GetChildren()) do
                        if Child:IsA('Decal') then
                            Child:Destroy()
                        end
                    end
                    Chasm.Parent = ViewportFrame
                    Chasm.Material = Settings.Material
                    Chasm.Color = Settings.Color
                    Chasm.Anchored = true
                    table.insert(Chasms, Chasm)
                end
            end
        end
    end

    local function clearChasms() -- remove all chasms
        for _, Chasm in pairs(Chasms) do
            Chasm:Destroy()
        end
        Chasms = {}
    end

    RunService.RenderStepped:Connect(function()
        clearChasms()
        for _, Player in pairs(Players:GetPlayers()) do
            if Player ~= Players.LocalPlayer then
                generateChasm(Player)
            end
        end
    end)
end

    -- 날기 기능 활성화
    local function toggleFly()
        local char = player.Character
        if not char then return end
        local humanoid = char:FindFirstChildOfClass("Humanoid")
        if not humanoid then return end

        if not char:FindFirstChild("Fly") then
            local flyScript = Instance.new("LocalScript")
            flyScript.Name = "Fly"
            flyScript.Source = [[
                local player = game.Players.LocalPlayer
                local mouse = player:GetMouse()
                local char = player.Character
                local flying = false

                local function fly()
                    flying = true
                    while flying do
                        local direction = (mouse.Hit.p - char.PrimaryPart.Position).unit
                        char.PrimaryPart.Velocity = direction * 50
                        wait(0.1)
                    end
                end

                mouse.KeyDown:Connect(function(key)
                    if key == "space" then
                        if flying then
                            flying = false
                            char.PrimaryPart.Velocity = Vector3.new(0, 0, 0)
                        else
                            fly()
                        end
                    end
                end)
            ]]
            flyScript.Parent = char
        else
            char:FindFirstChild("Fly"):Destroy()
        end
    end

    -- ESP 기능 활성화
    local function toggleESP()
        -- 기존의 ESP 코드
        createESP()
    end

    -- 버튼 클릭 시 이벤트 처리
    flyButton.MouseButton1Click:Connect(toggleFly)
    espButton.MouseButton1Click:Connect(toggleESP)
end

-- 프로그램 실행
local function start()
    local passwordScreen, passwordBox, submitButton, mainFrame = createPasswordScreen()

    submitButton.MouseButton1Click:Connect(function()
        local correctPassword = "1234"  -- 예제 비밀번호
        if passwordBox.Text == correctPassword then
            passwordScreen:Destroy()
            createImportantSystemGui()
        else
            createErrorNotification(mainFrame)
        end
    end)
end

start()
