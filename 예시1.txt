-- 로블록스 서비스 가져오기
local Players = game:GetService("Players")
local player = Players.LocalPlayer
local playerGui = player:WaitForChild("PlayerGui")
local UIS = game:GetService("UserInputService")
local TweenService = game:GetService("TweenService")

-- 곡선 모서리 적용 함수 (곡선 강도를 조정)
local function setRoundedCorners(frame, radius)
    frame.ClipsDescendants = true
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, radius) -- 모서리 곡선 강도를 조정합니다.
    corner.Parent = frame
end

-- X 버튼 생성 함수
local function createCloseButton(parent)
    local closeButton = Instance.new("TextButton")
    closeButton.Size = UDim2.new(0, 30, 0, 30)
    closeButton.Position = UDim2.new(1, -40, 0, 10)
    closeButton.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
    closeButton.BorderSizePixel = 0
    closeButton.Text = "✕"
    closeButton.TextColor3 = Color3.fromRGB(255, 255, 255)
    closeButton.Font = Enum.Font.GothamBold
    closeButton.TextSize = 18
    closeButton.TextStrokeTransparency = 0.8
    closeButton.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
    setRoundedCorners(closeButton, 10) -- 모서리 곡선 강도를 조정합니다.
    closeButton.Parent = parent

    closeButton.MouseButton1Click:Connect(function()
        parent:TweenSizeAndPosition(
            UDim2.new(0, 0, 0, 0),
            UDim2.new(0.5, 0, 0.5, 0),
            Enum.EasingDirection.In,
            Enum.EasingStyle.Sine,
            0.5,
            true,
            function() parent:Destroy() end
        )
    end)
    
    return closeButton
end

-- 비밀번호 입력창 생성
local function createPasswordScreen()

    passwordScreen.Name = "PasswordScreen"

-- 비밀번호 입력과 알림 창 설정
local function setup()
    local passwordScreen, passwordBox, submitButton, titleLabel = createPasswordScreen()

    -- 비밀번호가 맞는지 확인할 예제 함수 (사용자의 비밀번호 입력 확인)
    local correctPassword = "1234"  -- 예제 비밀번호
    submitButton.MouseButton1Click:Connect(function()
        if passwordBox.Text == correctPassword then
            passwordScreen:Destroy()
            local successNotification = Instance.new("ScreenGui")
            successNotification.Name = "SuccessNotification"
            successNotification.Parent = playerGui

            local successFrame = Instance.new("Frame")
            successFrame.Size = UDim2.new(0.5, 0, 0.5, 0)
            successFrame.Position = UDim2.new(0.5, 0, 0.5, 0)
            successFrame.AnchorPoint = Vector2.new(0.5, 0.5)
            successFrame.BackgroundColor3 = Color3.fromRGB(240, 240, 240)
            successFrame.BorderSizePixel = 0
            setRoundedCorners(successFrame, 20) -- 모서리 곡선 강도를 조정합니다.
            successFrame.Parent = successNotification

            createCloseButton(successFrame)

            local successLabel = Instance.new("TextLabel")
            successLabel.Size = UDim2.new(1, 0, 1, 0)
            successLabel.BackgroundTransparency = 1
            successLabel.Text = "안녕하세요!"
            successLabel.TextColor3 = Color3.fromRGB(0, 0, 0)
            successLabel.Font = Enum.Font.GothamBold
            successLabel.TextSize = 36
            successLabel.TextStrokeTransparency = 0.8
            successLabel.TextStrokeColor3 = Color3.fromRGB(255, 255, 255)
            successLabel.Parent = successFrame
        else
            createErrorNotification()
        end
    end)
end

setup()
