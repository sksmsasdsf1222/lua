-- 메인 프레임 생성
local function createMainFrame()
    local frame = Instance.new("Frame")
    frame.Size = mainFrameSize
    frame.Position = UDim2.new(0.5, -250, 0.4, -150) -- 화면 중앙에 배치
    frame.BackgroundColor3 = Color3.fromRGB(35, 35, 35)
    frame.BorderSizePixel = 0
    frame.Active = true
    frame.Draggable = true
    frame.Parent = playerGui:FindFirstChildOfClass("ScreenGui")

    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 10)
    corner.Parent = frame

    return frame
end

-- 결과 프레임 생성
local function createResultFrame()
    local frame = Instance.new("Frame")
    frame.Size = resultFrameSize
    frame.Position = UDim2.new(0.5, -resultFrameSize.X.Offset/2, 0.5, -resultFrameSize.Y.Offset/2)
    frame.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    frame.BorderSizePixel = 0
    frame.Visible = false
    frame.Parent = playerGui:FindFirstChildOfClass("ScreenGui")

    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 10)
    corner.Parent = frame

    local messageLabel = Instance.new("TextLabel")
    messageLabel.Size = UDim2.new(1, -20, 0.8, -20)
    messageLabel.Position = UDim2.new(0, 10, 0, 10)
    messageLabel.BackgroundTransparency = 1
    messageLabel.TextColor3 = Color3.fromRGB(0, 0, 0)
    messageLabel.Font = Enum.Font.SourceSans
    messageLabel.TextSize = 20
    messageLabel.TextWrapped = true
    messageLabel.Text = ""
    messageLabel.Parent = frame

    local closeButton = Instance.new("TextButton")
    closeButton.Size = UDim2.new(0, 30, 0, 30)
    closeButton.Position = UDim2.new(1, -40, 0, 10)
    closeButton.BackgroundColor3 = Color3.fromRGB(200, 50, 50)
    closeButton.Text = "X"
    closeButton.TextColor3 = Color3.fromRGB(255, 255, 255)
    closeButton.Font = Enum.Font.SourceSansBold

    closeButton.Parent = frame

    local corner2 = Instance.new("UICorner")
    corner2.CornerRadius = UDim.new(0, 8)
    corner2.Parent = closeButton

    closeButton.MouseButton1Click:Connect(function()
        frame.Visible = false
    end)

    return frame, messageLabel

-- 메인 프레임 및 UI 요소 생성
local screenGui = Instance.new("ScreenGui")
screenGui.Parent = playerGui

local mainFrame = createMainFrame()
local titleBar, closeButton = createTitleBar(mainFrame)
local keyInputBox, submitButton = createInputSection(mainFrame)
local resultFrame, messageLabel = createResultFrame()

-- 키 확인 및 결과 처리
local function handleKeySubmission()
    local inputKey = keyInputBox.Text

    if inputKey == adminKey or table.find(userKeys, inputKey) then
        -- 등록 완료 처리
        mainFrame.Visible = false
        resultFrame.Visible = true
        messageLabel.Text = "등록 완료\n당신은 관리자입니다. 절대로 키를 공유하지 마세요."
    else
        -- 잘못된 키 처리
        messageLabel.Text = "잘못된 키입니다."
    end
end

-- 버튼 클릭 이벤트
submitButton.MouseButton1Click:Connect(handleKeySubmission)
closeButton.MouseButton1Click:Connect(function()
    mainFrame:Destroy()
    resultFrame:Destroy()
end)

-- 게임이 종료될 때 자동으로 UI가 사라지게 하기 위한 설정
RunService.StarterGui:SetCoreGuiEnabled(Enum.CoreGuiType.All, false)
RunService.Heartbeat:Connect(function()
    if not playerGui:FindFirstChildOfClass("ScreenGui") then
        screenGui.Parent = playerGui
    end
end)
