local frame = script.Parent
local textBox = frame:WaitForChild("PasswordBox")
local submitButton = frame:WaitForChild("SubmitButton")
local screenGui = frame.Parent

-- 비밀번호 정의
local adminPassword = "admin"
local userPassword = "user"

-- 버튼 클릭 이벤트 처리
submitButton.MouseButton1Click:Connect(function()
    local inputText = textBox.Text
    local validPasswords = {adminPassword, userPassword}

    if table.find(validPasswords, inputText) then
        -- 비밀번호가 맞으면
        textBox.Text = ""
        frame.Visible = false

        -- 성공 메시지 Frame 생성
        local successFrame = Instance.new("Frame")
        successFrame.Name = "SuccessFrame"
        successFrame.Size = UDim2.new(0.4, 0, 0.4, 0)
        successFrame.Position = UDim2.new(0.3, 0, 0.3, 0)
        successFrame.BackgroundColor3 = Color3.fromRGB(0, 255, 0)
        successFrame.BorderSizePixel = 0
        successFrame.Parent = screenGui

        local successLabel = Instance.new("TextLabel")
        successLabel.Name = "SuccessLabel"
        successLabel.Text = "안녕"
        successLabel.Size = UDim2.new(1, 0, 1, 0)
        successLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
        successLabel.TextScaled = true
        successLabel.BackgroundTransparency = 1
        successLabel.Parent = successFrame

        -- 성공 메시지에 애니메이션 추가 (선택 사항)
        local tweenService = game:GetService("TweenService")
        local tweenInfo = TweenInfo.new(1, Enum.EasingStyle.Bounce, Enum.EasingDirection.InOut)
        local goal = {Position = UDim2.new(0.3, 0, 0.3, 0)}

        local tween = tweenService:Create(successFrame, tweenInfo, goal)
        tween:Play()
    else
        -- 비밀번호가 틀리면 (선택 사항)
        print("잘못된 비밀번호입니다.")
    end
end)
