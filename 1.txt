-- ScreenGui 생성
local screenGui = Instance.new("ScreenGui")
screenGui.Name = "MyScreenGui"
screenGui.Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui")

-- Frame 생성
local frame = Instance.new("Frame")
frame.Name = "MainFrame"
frame.Size = UDim2.new(0.4, 0, 0.4, 0)
frame.Position = UDim2.new(0.3, 0, 0.3, 0)
frame.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
frame.Parent = screenGui

-- Title Label 생성
local titleLabel = Instance.new("TextLabel")
titleLabel.Name = "TitleLabel"
titleLabel.Text = "[ KR ] Verifition"
titleLabel.Size = UDim2.new(1, 0, 0.2, 0)
titleLabel.Position = UDim2.new(0, 0, 0, 0)
titleLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
titleLabel.TextScaled = true
titleLabel.BackgroundTransparency = 1
titleLabel.Parent = frame

-- Password Input Box 생성
local textBox = Instance.new("TextBox")
textBox.Name = "PasswordBox"
textBox.PlaceholderText = "비밀번호를 입력하세요"
textBox.Size = UDim2.new(0.8, 0, 0.1, 0)
textBox.Position = UDim2.new(0.1, 0, 0.3, 0)
textBox.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
textBox.TextColor3 = Color3.fromRGB(0, 0, 0)
textBox.TextScaled = true
textBox.ClearTextOnFocus = true
textBox.Parent = frame

-- Submit Button 생성
local submitButton = Instance.new("TextButton")
submitButton.Name = "SubmitButton"
submitButton.Text = "확인"
submitButton.Size = UDim2.new(0.3, 0, 0.1, 0)
submitButton.Position = UDim2.new(0.35, 0, 0.45, 0)
submitButton.BackgroundColor3 = Color3.fromRGB(0, 122, 204)
submitButton.TextColor3 = Color3.fromRGB(255, 255, 255)
submitButton.TextScaled = true
submitButton.Parent = frame

-- Success Label 생성
local successLabel = Instance.new("TextLabel")
successLabel.Name = "SuccessLabel"
successLabel.Text = "안녕"
successLabel.Size = UDim2.new(1, 0, 0.2, 0)
successLabel.Position = UDim2.new(0, 0, 0.6, 0)
successLabel.TextColor3 = Color3.fromRGB(0, 255, 0)
successLabel.TextScaled = true
successLabel.BackgroundTransparency = 1
successLabel.Visible = false
successLabel.Parent = frame

-- Error Label 생성 (선택 사항)
local errorLabel = Instance.new("TextLabel")
errorLabel.Name = "ErrorLabel"
errorLabel.Text = "잘못된 비밀번호입니다."
errorLabel.Size = UDim2.new(1, 0, 0.2, 0)
errorLabel.Position = UDim2.new(0, 0, 0.6, 0)
errorLabel.TextColor3 = Color3.fromRGB(255, 0, 0)
errorLabel.TextScaled = true
errorLabel.BackgroundTransparency = 1
errorLabel.Visible = false
errorLabel.Parent = frame

-- 버튼 클릭 이벤트 처리
submitButton.MouseButton1Click:Connect(function()
    local inputText = textBox.Text
    local adminPassword = "admin"
    local userPassword = "user"

    if inputText == adminPassword or inputText == userPassword then
        -- 비밀번호가 맞으면
        textBox.Text = ""
        successLabel.Visible = true
        errorLabel.Visible = false
    else
        -- 비밀번호가 틀리면
        successLabel.Visible = false
        errorLabel.Visible = true
    end
end)
